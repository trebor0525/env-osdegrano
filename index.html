import pulp # Oferta, demanda y costos oferta = {"Manizales": 100, "Pereira": 150, "Armenia": 95} demanda = {"Sarajevo": 75, "Ucrania": 150, "Croacia": 120} # Costos definidos con strings costos = { "Manizales-Sarajevo": 4, "Manizales-Ucrania": 6, "Manizales-Croacia": 8, "Pereira-Sarajevo": 5, "Pereira-Ucrania": 4, "Pereira-Croacia": 3, "Armenia-Sarajevo": 9, "Armenia-Ucrania": 7, "Armenia-Croacia": 4, } # Nodos de origen y destino nodos_origen = list(oferta.keys()) nodos_destino = list(demanda.keys()) # Modelo modelo = pulp.LpProblem("Transporte_de_grano", pulp.LpMinimize) # Variables x = pulp.LpVariable.dicts("x", (i + '-' + j for i in nodos_origen for j in nodos_destino), lowBound=0) # Funci√≥n objetivo modelo += pulp.lpSum(costos[i + '-' + j] * x[i + '-' + j] for i in nodos_origen for j in nodos_destino) # Restricciones de oferta for i in nodos_origen: modelo += pulp.lpSum(x[i + '-' + j] for j in nodos_destino) <= oferta[i] # Restricciones de demanda for j in nodos_destino: modelo += pulp.lpSum(x[i + '-' + j] for i in nodos_origen) >= demanda[j] # Resolver el modelo modelo.solve() # Resultados print("Plan de transporte √≥ptimo:") for v in modelo.variables(): if v.varValue > 0: print(f"{v.name} = {v.varValue}") print(f"Costo total: {pulp.value(modelo.objective)}")
